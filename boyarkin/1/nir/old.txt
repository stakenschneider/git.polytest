2) 
Comple *.so library 
~$ clang -shared -o libfilename.so -fPIC filename.c

*.cpp to *.ll 
~$ clang++ -fms-compatibility-version=19 filename.cpp -emit-llvm -S

*.c to *.bc
~$ clang -c filename.c -emit-llvm -o filename.bc

*.ll to *.bc 
~$ llvm-as filename.ll -o filename.bc

*.bc to *.ll 
~$ llvm-dis filename.bc -o filename.ll





2) python ffi usage 

#EXAMPLE
	import ctypes

	def main():
		stringtools = ctypes.CDLL("./libcfile.so")
		print(stringtools.count_substrings(b"banana", b"na"))
		print 'Hello, world!'
	 
	if __name__ == '__main__':
		main()
#END

~$ python -m test test.py 




3) ruby ffi usage

#EXAMPLE
	require 'ffi'

	module Hello
	  extend FFI::Library
	  ffi_lib './libcfile.so'
	  attach_function :foo, [], :int  # [:arg1, :arg2] :ret
	end

	Hello.foo
#END

~$ ruby test.rb
